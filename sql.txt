CREATE TABLE Department (
    dept_id INT AUTO_INCREMENT PRIMARY KEY,  -- unique ID (auto generated)
    dept_name VARCHAR(100) UNIQUE NOT NULL,  -- department name must be unique
    location VARCHAR(100),                   -- where the department is located
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP -- when it was created
);
